Program.cs 
// 1. 語音 → 逐字稿
var transcript = await whisper
  .TranscribeAsync(audioFile);

// 2. 逐字稿 → 會議記錄
var minutes = await openai
  .GenerateMinutesAsync(transcript);

// 3. 會議記錄 → Word 檔
var doc = new MeetingDocument();
doc.Build(minutes);
doc.SaveAs("meeting.docx");
