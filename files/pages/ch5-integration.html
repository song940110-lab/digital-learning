C# — 語音檔輸入流程 
// 1. Whisper 語音 → 逐字稿
var whisperService = new WhisperService(apiKey);
string transcript = await whisperService.TranscribeAsync("meeting.mp3");
Console.WriteLine($"✅ Whisper 轉譯完成：{transcript.Length} 字");

// 2. 儲存逐字稿（方便後續查閱）
await File.WriteAllTextAsync("transcript.txt", transcript);

// 3. GPT 整理 → Word（與逐字稿流程相同）
var openAIService = new OpenAIService(apiKey);
string minutes = await openAIService.GenerateMinutesAsync(transcript);

var docService = new DocumentService();
docService.BuildDocument(minutes, "output.docx");
Console.WriteLine("✅ 完成！檔案儲存於 output.docx");

4完整 Program.cs

以下是整合兩種輸入模式的主程式，自動判斷輸入類型並執行對應流程：

C# — Program.cs 完整整合版 
using OpenAI;
using OpenAI.Audio;
using OpenAI.Chat;
using DocumentFormat.OpenXml;
using DocumentFormat.OpenXml.Packaging;
using DocumentFormat.OpenXml.Wordprocessing;

var apiKey = Environment.GetEnvironmentVariable("OPENAI_API_KEY")
    ?? throw new Exception("請先設定 OPENAI_API_KEY 環境變數");

// ======= 輸入設定 =======
string inputFile  = "meeting.mp3";  // 或 "transcript.txt"
string outputFile = "output.docx";
bool   isAudio    = inputFile.EndsWith(".mp3") || inputFile.EndsWith(".m4a")
                   || inputFile.EndsWith(".wav") || inputFile.EndsWith(".mp4");

// ======= Step 1: 取得逐字稿 =======
string transcript;
if (isAudio)
{
    Console.WriteLine("🎙 語音檔偵測到，正在呼叫 Whisper...");
    var audioClient = new AudioClient("whisper-1", apiKey);
    using var stream = File.OpenRead(inputFile);
    var result = await audioClient.TranscribeAudioAsync(
        stream, inputFile,
        new AudioTranscriptionOptions { Language = "zh" }
    );
    transcript = result.Text;
    Console.WriteLine($"✅ Whisper 完成：{transcript.Length} 字");
}
else
{
    Console.WriteLine("📝 讀取逐字稿檔案...");
    transcript = await File.ReadAllTextAsync(inputFile);
    Console.WriteLine($"✅ 逐字稿讀取完成：{transcript.Length} 字");
}

// ======= Step 2: GPT 生成會議記錄 =======
Console.WriteLine("✨ 正在生成會議記錄...");
var chatClient = new ChatClient("gpt-4o-mini", apiKey);
var systemPrompt = """
    你是專業會議記錄整理助理，請依以下格式輸出，嚴格遵守格式，不要加入額外說明：
    【會議主題】：
    【日期時間】：
    【出席人員】：
    【討論事項】：
    【結論與決議】：
    【待辦事項】（表格格式：負責人|項目|截止日）：
    """;

var messages = new[] {
    new SystemChatMessage(systemPrompt),
    new UserChatMessage(transcript)
};
var completion = await chatClient.CompleteChatAsync(
    messages,
    new ChatCompletionOptions { MaxOutputTokenCount = 2000 }
);
string minutes = completion.Content[0].Text;
Console.WriteLine("✅ 會議記錄生成完成");

// ======= Step 3: 輸出 Word 文件 =======
Console.WriteLine("📄 正在建立 Word 文件...");
File.Copy("template.dotx", outputFile, overwrite: true);

using var doc = WordprocessingDocument.Open(outputFile, true);
doc.ChangeDocumentType(WordprocessingDocumentType.Document);
var body = doc.MainDocumentPart!.Document.Body!;
body.RemoveAllChildren<Paragraph>();

// 將會議記錄逐行填入
foreach (var line in minutes.Split('\n'))
{
    var isBold = line.StartsWith("【");
    var para = new Paragraph();
    var run  = new Run();
    if (isBold) run.AppendChild(new RunProperties(new Bold()));
    run.AppendChild(new Text(line) { Space = SpaceProcessingModeValues.Preserve });
    para.AppendChild(run);
    body.AppendChild(para);
}

doc.MainDocumentPart.Document.Save();
Console.WriteLine($"🎉 完成！會議記錄已儲存至：{outputFile}");

5常見錯誤處理

錯誤訊息 原因 解決方式
401 Unauthorized API Key 錯誤或未設定 確認環境變數 OPENAI_API_KEY 正確設定
429 Rate Limit 請求太頻繁或餘額不足 等待幾秒後重試，或檢查 OpenAI 帳號餘額
File too large 音訊檔超過 25MB 用 FFmpeg 壓縮音訊至 25MB 以下
InvalidOperationException Word 模板路徑錯誤 確認 template.dotx 在執行目錄下
輸出空白 Word 文件 GPT 回傳空字串 確認逐字稿不是空的，並檢查 Prompt 是否正確

加入 try-catch 保護

C# — 錯誤處理範例 
try
{
    var result = await chatClient.CompleteChatAsync(messages);
    string minutes = result.Content[0].Text;
    if (string.IsNullOrWhiteSpace(minutes))
        throw new Exception("AI 回傳空白內容，請檢查逐字稿");
}
catch (Exception ex)
{
    Console.WriteLine($"❌ 錯誤：{ex.Message}");
    // 寫入 log 檔案方便除錯
    await File.AppendAllTextAsync("error.log",
        $"{DateTime.Now}: {ex}\n");
}

6完整 Demo 步驟

按照以下步驟，從頭到尾跑一次完整流程：

1

準備輸入檔案

準備一個 MP3 錄音檔 meeting.mp3，或一個逐字稿文字檔 transcript.txt，放在專案的執行目錄下（bin/Debug/net8.0/）。

2

確認環境變數

確認 OPENAI_API_KEY 已正確設定，在 PowerShell 執行： 

echo $env:OPENAI_API_KEY
# 應該顯示 sk-proj-xxx...
3

在 Program.cs 設定輸入檔名

修改 inputFile 變數為你的檔案名稱，確認 isAudio 判斷正確。

4

執行程式（F5 或 Ctrl+F5）

觀察主控台輸出，依序看到：

🎙 語音檔偵測到，正在呼叫 Whisper...
✅ Whisper 完成：3214 字
✨ 正在生成會議記錄...
✅ 會議記錄生成完成
📄 正在建立 Word 文件...
🎉 完成！會議記錄已儲存至：output.docx
5

開啟 output.docx 確認結果

用 Word 或 LibreOffice 開啟輸出的 output.docx，確認會議記錄內容正確、格式整齊。
